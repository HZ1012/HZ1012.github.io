### 树

树是一种数据结构，比如：目录结构

树是一种可以递归定义的数据结构

树是由n个节点组成的集合：

* 如果n=0，那这是一棵空树
* 如果n>0，那存在1个节点作为树的根节点，其他节点可以分为m个集合，每个集合本身又是一棵树

#### 二叉树

二叉树的链式存储：将二叉树的节点定义为一个对象，节点之间通过类似链表的链接方式来连接

二叉树的遍历方式：

前序遍历，中序遍历，后序遍历，层次遍历

#### 二叉搜索树

二叉搜索是一棵二叉树且满足性质：设x是二叉树的一个节点。如果y是x的左子树的一个节点，那么y.key$\leq$ x.key；如果y是x的右子树的一个节点，那么y.key$\geq$ x.key

二叉搜索树的操作：查询、插入、删除

删除操作：

* 如果要删除的节点是叶子节点，直接删除
* 如果要删除的节点只有一个孩子：将此节点的父亲与孩子连接，然后删除该节点
* 如果要删除的节点有两个孩子：将其右子树的最小节点删除，并替换当前节点

二叉搜索树的效率：

平均情况下，二叉搜索树进行搜索的时间复杂度为$O(logn)$

最坏情况下：二叉搜索树可能非常偏斜

解决方案：随机化插入，AVL树

#### AVL树

AVL是一棵自平衡的二叉搜索树

性质：

根的左右子树的高度差的绝对值不能超过1

根的左右子树都是平衡二叉树

**AVL树插入**

插入一个节点可能会破坏AVL树的平衡，可以通过旋转操作来修正

插入一个节点后，只有从插入节点到根节点的路径上的节点的平衡可能改变，称之为K，K的两棵子树的高度差为2

不平衡有四种情况：

1. 不平衡是由于对K的右孩子的右子树插入导致的：左旋
2. 不平衡是由于对K的左孩子的左子树插入导致的·：右旋
3. 不平衡是由于对K的右孩子的左子树插入导致的：右旋——左旋
4. 不平衡是由于对K的左孩子的右子树插入导致的：左旋——右旋

**二叉搜索树扩展应用——B树**

B-Tree：是一棵自平衡的多路搜索树，常用于数据库的索引。

class Node:
    def __init__(self,name,type = 'dir'):
        self.name = name
        self.type = type # 'dir'or'type'
        self.children = []
        self.parent = None
    
    def __repr__(self):
        return self.name
    
class FileSystemTree:
    def __init__(self):
        self.root = Node('/')
        self.now = self.root
        
    def mkdir(self,name):
        # name以/结尾
        if name[-1]!='/':
            name+='/'
        node = Node(name)
        self.now.children.append(node)
        node.parent=self.now
    
    def ls(self):
        return self.now.children
    
    def cd(self,name):
        if name[-1]!='/':
            name+='/'
        if name == '../': # 用../回到上一级
            self.now = self.now.parent
            return # 到这一步不再执行函数
        for child in self.now.children:
            if child.name == name:
                self.now = child
                return
        raise ValueError('invalid dir')

class BiTreeNode:
    def __init__(self,data):
        self.data =data

        self.rchild=None
        self.parent=None

class BST:
    def __init__(self,li=None):
        self.root =None
        if li:
            for val in li:
                self.insert_no_rec(val) #生成二叉搜索树
        
    def insert(self,node,val):  
        if not node:
            node =BiTreeNode(val)
        elif val<node.data:
            node.lchild=self.insert(node.lchild, val)
            node.lchild.parent = node
        elif val>node.data:
            node.rchild=self.insert(node.rchild, val)
            node.rchild.parent = node
        return node
    
    def insert_no_rec(self,val):
        p=self.root
        if not p:
            self.root = BiTreeNode(val)
            return
        while True:
            if val<p.data:
                if p.lchild:
                    p=p.lchild
                else:
                    p.lchild =BiTreeNode(val)
                    p.lchild.parent = p
                    return
            elif val>p.data:
                if p.rchild:
                    p=p.rchild
                else:
                    p.rchild = BiTreeNode(val)
                    p.rchild.parent = p
                    return
            else:
                return
    
    def query(self,node,val): # 递归查询
        if not node:
            return None
        if val>node.data:
            return self.query(node.rchild, val)
        elif val<node.data:
            return self.query(node.lchild, val)
        else:
            return node
    def query_no_rec(self,val): #非递归查询
        p=self.root
        while p:
            if p.data<val:
                p=p.rchild 
            elif p.data>val:
                p=p.lchild 
            else:
                return p
        return None
                
    
    
    def pre_order(self,root): # 前序遍历
        if root:
            print(root.data,end=',')
            self.pre_order(root.lchild)
            self.pre_order(root.rchild)# 递归调用


    def in_order(self,root): # 中序遍历
        if root:
            self.in_order(root.lchild)
            print(root.data,end=',')
            self.in_order(root.rchild)

    def post_order(self,root): # 后序遍历

            self.post_order(root.lchild)
            self.post_order(root.rchild)

    def __remove_node_1(self,node): # 删除的自调用函数
        # 情况1:node是叶子节点
        if not node.parent:
            self.root = None
        if node== node.parent.lchild:
            node.parent.lchild =None
        else:
            node.parent.rchild =None

            node.lchild.parent = None

            node.lchild.parent = node.parent
        else:
            node.parent.rchild=node.lchild
            node.lchild.parent = node.parent

            self.root = node.rchild
            node.rchild.parent=None

            node.parent.lchild=node.rchild
            node.rchild.parent = node.parent
        else:
            node.parent.rchild=node.rchild
            node.rchild.parent = node.parent

            node = self.query_no_rec(val)
            if not node: 
                return False

                self.__remove_node_1(node)

                self.__remove_node_21(node)

                self.__remove_node_22(node)

                    min_node = min_node.lchild
                node.data = min_node.data
                if min_node.rchild:
                    self.__remove_node_22(min_node)
                else:
                    self.__remove_node_1(min_node)

#%% AVL
class AVLNode():
    def __init__(self,data):
        BiTreeNode.__init__(self,data)
        self.bf = 0

class AVLTree(BST):
    def __init__(self,li=None):
        BST.__init__(self,li)
    
    def rotate_left(self,p,c):
        s2 = c.lchild
        p.rchild = s2
        if s2: # 判断是否非空，非空时变换父亲节点
            s2.parent = p
        c.lchild = p
        p.parent = c
        # 更新blance factor
        p.bf = 0
        c.bf = 0
        return c
        
    def rotate_right(self,p,c):
        s2 = c.rchild
        p.lchild = s2
        if s2:
            s2.parent =p
        c.rchild = p
        p.parent = c
        # 更新blance factor
        p.bf = 0
        c.bf = 0
        return c
    
    def rotate_right_left(self,p,c):
        # 先右旋
        g = c.lchild
        s3 = g.rchild
        c.lchild = s3
        if s3:
            s3.parent = c
        g.rchild = c
        c.parent = g
        # 再左旋
        s2 = g.lchild
        p.rchild = s2
        if s2:
            s2.parent = p
        g.lchild = p
        p.parent = g
        # 更新blance factor
        if g.bf>0:
            p.bf = -1
            c.bf = 0
        else:
            p.bf = 0 
            c.bf = 1
        g.bf = 0
        return g
    
    def rotate_left_right(self,p,c):
        # 先左旋
        g = c.rchild
        s2 = g.lchild
        c.rchild = s2
        if s2:
            s2.parent = c
        g.lchild = c
        c.parent = g
        # 再右旋
        s3 = g.rchild
        p.lchild = s3
        if s3:
            s3.parent = p
        g.rchild = p
        p.parent =g
        # 更新blance factor
        if g.bf<0:
            p.bf = 1
            c.bf = 0
        else:
            p.bf = 0
            c.bf = -1
        g.bf = 0
        return g
    
    def insert_no_rec(self,val):
        # step1:和二叉搜索树一样先插入
        p=self.root
        if not p:
            self.root = AVLNode(val)
            return
        while True:
            if val<p.data:
                if p.lchild:
                    p=p.lchild
                else:
                    p.lchild = AVLNode(val)
                    p.lchild.parent = p
                    node = p.lchild # node存储的就是插入的结点
                    break
            elif val>p.data:
                if p.rchild:
                    p=p.rchild
                else:
                    p.rchild = AVLNode(val)
                    p.rchild.parent = p
                    node = p.rchild
                    break
            else: # val == p.data
                return
        # step2:更新blance factor
        while node.parent: # 保证node.parent 非空
            if node.parent.lchild == node: # 插入到左子树 bf-=1
                if node.parent.bf<0:#-1到-2,需旋转
                    # 看node哪边沉,旋转后重新连接
                    g = node.parent.parent #连接旋转后的子树
                    x = node.parent # 旋转前子树的根
                    if node.bf>0:
                        n = self.rotate_left_right(node.parent, node)
                    else:
                        n = self.rotate_right(node.parent, node)
                elif node.parent.bf>0: #1到0,不需要旋转
                    node.parent.bf = 0
                    break # 不平衡不再传递
                else: #0到-1,不需要旋转，继续传递
                    node.parent.bf = -1
                    node = node.parent
                    continue
            else: # 插入到右子树 bf+=1
                if node.parent.bf>0:
                    g = node.parent.parent #连接旋转后的子树
                    x = node.parent # 旋转前子树的根
                    if node.bf<0:
                        n = self.rotate_right_left(node.parent, node)
                    else:
                        n = self.rotate_left(node.parent, node)
                elif node.parent.bf<0:
                    node.parent.bf = 0
                    break
                else:
                    node.parent.bf = 1
                    node = node.parent
                    continue
            # 连接旋转后的子树
            n.parent = g
            if g:
                if x == g.lchild:
                    g.lchild = n
                else:
                    g.rchild = n
                break
            else:
                self.root =n
                break
